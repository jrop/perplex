    Lexer

    Lexes a source-string into tokens.

    Example:

    const lex = perplex('...')
  .token('ID', /my-id-regex/)
  .token('(', /\(/)
  .token(')', /\)/)
  .token('WS', /\s+/, true) // true means 'skip'

while ((let t = lex.next()).type != 'EOF') {
  console.log(t)
}
// alternatively:
console.log(lex.toArray())
